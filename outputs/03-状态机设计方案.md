# 订单状态机设计方案

> 版本：v1.0  
> 日期：2026-04-30  

---

## 一、方案选型结论

采用**轻量级自研状态机**，基于"转移表 + 策略模式"实现。

核心理念：
- 状态转移规则用代码定义（类型安全、编译期检查）
- 前置条件和动作通过策略模式注入（灵活可扩展）
- 每次转移自动写变更日志（不可遗漏）
- 生产订单行支持跳过特定环节（可配置）

不选 Spring StateMachine 的原因：功能过剩、调试困难、与业务对象绑定太重。

### 2.2 转移定义

```java
/**
 * 状态转移定义（不可变对象）
 */
public class Transition<S, E> {
    
    private final S source;           // 源状态
    private final S target;           // 目标状态
    private final E event;            // 触发事件
    private final String description; // 描述
    private final Condition<S, E> condition;  // 前置条件（可选）
    private final Action<S, E> action;        // 转移动作（可选）

    // builder 模式构造
    public static <S, E> Builder<S, E> from(S source) {
        return new Builder<>(source);
    }

    public static class Builder<S, E> {
        private S source;
        private S target;
        private E event;
        private String description;
        private Condition<S, E> condition;
        private Action<S, E> action;

        public Builder(S source) { this.source = source; }

        public Builder<S, E> to(S target) { this.target = target; this.event = event; return this; }
        public Builder<S, E> on(E event) { this.event = event; return this; }
        public Builder<S, E> desc(String description) { this.description = description; return this; }
        public Builder<S, E> when(Condition<S, E> condition) { this.condition = condition; return this; }
        public Builder<S, E> then(Action<S, E> action) { this.action = action; return this; }
        public Transition<S, E> build() {
            return new Transition<>(source, target, event, description, condition, action);
        }
    }
}
```

## 三、状态机引擎

### 3.1 核心引擎

```java
/**
 * 通用状态机引擎
 * 线程安全，无状态，可单例使用
 */
public class StateMachine<S, E> {

    private final String machineId;
    private final Map<S, Map<E, Transition<S, E>>> transitionTable;  // source → (event → transition)
    private final List<TransitionListener<S, E>> listeners = new CopyOnWriteArrayList<>();

    private StateMachine(String machineId, List<Transition<S, E>> transitions) {
        this.machineId = machineId;
        this.transitionTable = new HashMap<>();
        for (Transition<S, E> t : transitions) {
            transitionTable
                .computeIfAbsent(t.getSource(), k -> new HashMap<>())
                .put(t.getEvent(), t);
        }
    }

    /**
     * 触发事件，尝试状态转移
     * @return 转移后的新状态
     * @throws StateMachineException 如果转移不合法
     */
    public S fireEvent(S currentState, E event, TransitionContext<S, E> context) {
        // 1. 查找转移定义
        Map<E, Transition<S, E>> eventMap = transitionTable.get(currentState);
        if (eventMap == null) {
            throw new StateMachineException(
                String.format("[%s] 当前状态 %s 不允许任何转移", machineId, currentState)
            );
        }
        Transition<S, E> transition = eventMap.get(event);
        if (transition == null) {
            throw new StateMachineException(
                String.format("[%s] 当前状态 %s 不允许事件 %s", machineId, currentState, event)
            );
        }

        // 2. 前置条件校验
        if (transition.getCondition() != null) {
            context.setCurrentState(currentState);
            context.setTargetState(transition.getTarget());
            context.setEvent(event);
            if (!transition.getCondition().evaluate(context)) {
                throw new StateMachineException(
                    String.format("[%s] 状态转移 %s → %s 的前置条件不满足", 
                        machineId, currentState, transition.getTarget())
                );
            }
        }

        S targetState = transition.getTarget();

        // 3. 前置通知（beforeTransition）
        listeners.forEach(l -> l.beforeTransition(currentState, targetState, event, context));

        // 4. 执行转移动作
        if (transition.getAction() != null) {
            transition.getAction().execute(context);
        }

        // 5. 后置通知（afterTransition）
        listeners.forEach(l -> l.afterTransition(currentState, targetState, event, context));

        return targetState;
    }

    /**
     * 查询当前状态允许的下一个状态列表（前端可以用这个来控制按钮显隐）
     */
    public List<Transition<S, E>> getAvailableTransitions(S currentState) {
        Map<E, Transition<S, E>> eventMap = transitionTable.get(currentState);
        if (eventMap == null) {
            return Collections.emptyList();
        }
        return new ArrayList<>(eventMap.values());
    }

    /**
     * 注册监听器
     */
    public void addListener(TransitionListener<S, E> listener) {
        listeners.add(listener);
    }

    /**
     * 构建器
     */
    public static <S, E> StateMachineBuilder<S, E> builder(String machineId) {
        return new StateMachineBuilder<>(machineId);
    }
}
```

```java
public class StateMachineBuilder<S, E> {

        this.machineId = machineId;
```

## 四、销售订单状态机配置

```java
/**
 * 销售订单状态机配置
 */
@Configuration
public class SalesOrderStateMachineConfig {

    @Bean
    public StateMachine<SalesOrderStatus, SalesOrderEvent> salesOrderStateMachine(
            SalesOrderConditionEvaluator conditionEvaluator,
            SalesOrderActionExecutor actionExecutor,
            OrderChangeLogService changeLogService) {

        StateMachine<SalesOrderStatus, SalesOrderEvent> sm = 
            StateMachine.<SalesOrderStatus, SalesOrderEvent>builder("SALES_ORDER")
                
                // DRAFT → PENDING_APPROVAL：提交审批
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(DRAFT)
                    .to(PENDING_APPROVAL)
                    .on(SUBMIT)
                    .desc("提交订单审批")
                    .when(ctx -> conditionEvaluator.hasOrderLines(ctx))
                    .build())

                // PENDING_APPROVAL → CONFIRMED：审批通过
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(PENDING_APPROVAL)
                    .to(CONFIRMED)
                    .on(APPROVE)
                    .desc("审批通过")
                    .then(ctx -> actionExecutor.onOrderConfirmed(ctx))         // 触发库存预留
                    .build())

                // PENDING_APPROVAL → REJECTED：审批驳回
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(PENDING_APPROVAL)
                    .to(REJECTED)
                    .on(REJECT)
                    .desc("审批驳回")
                    .build())

                // REJECTED → PENDING_APPROVAL：修改后重新提交
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(REJECTED)
                    .to(PENDING_APPROVAL)
                    .on(RESUBMIT)
                    .desc("修改后重新提交")
                    .when(ctx -> conditionEvaluator.hasOrderLines(ctx))
                    .build())

                // DRAFT → CANCELLED：取消
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(DRAFT)
                    .to(CANCELLED)
                    .on(CANCEL)
                    .desc("取消订单")
                    .build())

                // CONFIRMED → PRODUCING：开始排产
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(CONFIRMED)
                    .to(PRODUCING)
                    .on(START_PRODUCE)
                    .desc("开始排产")
                    .when(ctx -> conditionEvaluator.hasLinkedProductionOrder(ctx))  // 已关联生产订单
                    .then(ctx -> actionExecutor.onOrderProducing(ctx))              // 通知采购模块
                    .build())

                // CONFIRMED → CANCELLED：取消（仅未关联生产订单时）
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(CONFIRMED)
                    .to(CANCELLED)
                    .on(CANCEL)
                    .desc("取消订单")
                    .when(ctx -> conditionEvaluator.hasNoLinkedProductionOrder(ctx))
                    .then(ctx -> actionExecutor.onOrderCancelled(ctx))             // 释放库存预留
                    .build())

                // PRODUCING → READY：备货完成
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(PRODUCING)
                    .to(READY)
                    .on(READY_STOCK)
                    .desc("备货完成")
                    .when(ctx -> conditionEvaluator.allStockFulfilled(ctx))        // 所有SKU库存满足
                    .build())

                // PRODUCING → SHIPPED：部分发货（允许部分发货场景）
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(PRODUCING)
                    .to(SHIPPED)
                    .on(SHIP)
                    .desc("部分发货")
                    .when(ctx -> conditionEvaluator.partialStockFulfilled(ctx))
                    .build())

                // READY → SHIPPED：发货
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(READY)
                    .to(SHIPPED)
                    .on(SHIP)
                    .desc("发货")
                    .when(ctx -> conditionEvaluator.hasOutboundOrder(ctx))         // 已创建出库单
                    .build())

                // SHIPPED → COMPLETED：签收
                .withTransition(Transition.<SalesOrderStatus, SalesOrderEvent>from(SHIPPED)
                    .to(COMPLETED)
                    .on(SIGN_OFF)
                    .desc("确认签收")
                    .build())

                .build();

        // 注册监听器：自动写变更日志
        sm.addListener(new TransitionListener<>() {
            @Override
            public void afterTransition(SalesOrderStatus from, SalesOrderStatus to, 
                                        SalesOrderEvent event, TransitionContext<SalesOrderStatus, SalesOrderEvent> ctx) {
                changeLogService.log(
                    "SALES", ctx.getOrderId(), ctx.getOrderLineId(),
                    "STATUS_CHANGE", "status", from.name(), to.name(),
                    event.name(), ctx.getOperatorId()
                );
            }
        });

        return sm;
    }
}
```

## 五、生产订单状态机配置（含跳过逻辑）

```java
/**
 * 生产订单状态机配置
 */
@Configuration
public class ProductionOrderStateMachineConfig {

    @Bean
    public StateMachine<ProductionOrderStatus, ProductionOrderEvent> productionOrderStateMachine(
            ProductionOrderConditionEvaluator conditionEvaluator,
            ProductionOrderActionExecutor actionExecutor,
            OrderChangeLogService changeLogService) {

        return StateMachine.<ProductionOrderStatus, ProductionOrderEvent>builder("PRODUCTION_ORDER")

                // DRAFT → RELEASED
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(DRAFT)
                    .to(RELEASED).on(RELEASE)
                    .desc("下达生产订单")
                    .when(ctx -> conditionEvaluator.hasBomAndQuantity(ctx))
                    .build())

                // RELEASED → PLANNED
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(RELEASED)
                    .to(PLANNED).on(PLAN)
                    .desc("排产完成")
                    .build())

                // PLANNED → CUTTING（需要检查是否跳过裁剪）
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(PLANNED)
                    .to(CUTTING).on(START_CUTTING)
                    .desc("开始裁剪")
                    .when(ctx -> !conditionEvaluator.skipCutting(ctx))   // 不跳过裁剪时才能走这条路
                    .build())

                // PLANNED → SEWING（跳过裁剪时直接到缝制）
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(PLANNED)
                    .to(SEWING).on(START_SEWING)
                    .desc("跳过裁剪，直接缝制")
                    .when(ctx -> conditionEvaluator.skipCutting(ctx))    // 跳过裁剪时走这条路
                    .build())

                // CUTTING → SEWING
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(CUTTING)
                    .to(SEWING).on(START_SEWING)
                    .desc("裁剪完成，进入缝制")
                    .build())

                // SEWING → FINISHING
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(SEWING)
                    .to(FINISHING).on(START_FINISHING)
                    .desc("缝制完成，进入后整")
                    .build())

                // FINISHING → COMPLETED
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(FINISHING)
                    .to(COMPLETED).on(COMPLETE)
                    .desc("生产完工")
                    .then(ctx -> actionExecutor.onProductionCompleted(ctx))  // 通知库存模块准备入库
                    .build())

                // COMPLETED → STOCKED
                .withTransition(Transition.<ProductionOrderStatus, ProductionOrderEvent>from(COMPLETED)
                    .to(STOCKED).on(STOCK_IN)
                    .desc("入库完成")
                    .when(ctx -> conditionEvaluator.allStockedIn(ctx))
                    .then(ctx -> actionExecutor.onProductionStocked(ctx))    // 通知销售订单可以发货
                    .build())

                .build();
    }
}
```

**跳过裁剪的关键设计：** 同一个源状态（PLANNED）对同一个事件，通过 **不同的条件** 走向不同的目标状态。`START_SEWING` 事件在 `skipCutting=true` 时直接从 PLANNED 到 SEWING，在 `skipCutting=false` 时不允许（必须先走 START_CUTTING）。

## 六、Service 层如何调用

```java
@Service
public class SalesOrderDomainService {

    private final StateMachine<SalesOrderStatus, SalesOrderEvent> stateMachine;
    private final SalesOrderRepository salesOrderRepository;

    /**
     * 触发状态转移
     */
    @Transactional
    public void fireEvent(Long orderId, SalesOrderEvent event, Long operatorId) {
        // 1. 加载订单
        SalesOrder order = salesOrderRepository.findById(orderId)
            .orElseThrow(() -> new BizException("订单不存在"));

        // 2. 构建上下文
        TransitionContext<SalesOrderStatus, SalesOrderEvent> context = 
            new TransitionContext<>(orderId, order.getStatus(), null, event, operatorId);

        // 3. 执行状态转移（引擎内部会：校验条件 → 执行动作 → 通知监听器）
        SalesOrderStatus newStatus = stateMachine.fireEvent(order.getStatus(), event, context);

        // 4. 更新订单状态
        order.setStatus(newStatus);
        salesOrderRepository.save(order);
    }

    /**
     * 查询当前可执行的操作（前端据此渲染按钮）
     */
    public List<AvailableActionVO> getAvailableActions(Long orderId) {
        SalesOrder order = salesOrderRepository.findById(orderId)
            .orElseThrow(() -> new BizException("订单不存在"));
        
        List<Transition<SalesOrderStatus, SalesOrderEvent>> transitions = 
            stateMachine.getAvailableTransitions(order.getStatus());

        return transitions.stream()
            .map(t -> new AvailableActionVO(
                t.getEvent().name(),
                t.getDescription(),
                t.getTarget().name()
            ))
            .collect(Collectors.toList());
    }
}
```

**前端使用场景：** 订单详情页调用 `GET /api/sales-orders/{id}/available-actions`，拿到当前可执行的操作列表，动态渲染按钮。比如订单在 DRAFT 状态时显示"提交"和"取消"按钮，在 CONFIRMED 状态时显示"排产"和"取消"按钮。

```java
        return order != null && !order.getLines().isEmpty();
    }
```

## 八、为什么不用数据库配置？

有人可能问：为什么不把状态转移规则存数据库，让运营可以自己配？

**因为状态转移是业务核心逻辑，不是配置。** 改一个状态流转规则可能意味着：
- 前端按钮逻辑要变
- 前置条件变了，校验逻辑要重写
- 动作变了，下游影响要评估

这些不是改一条数据库记录就能安全完成的，必须走完整的开发-测试-发布流程。把规则放在代码里，改了就要过 code review、过测试、过发布，这是正确的行为。

**数据库配置适用的场景：** 审批流的节点和条件（比如"超过10万需要总监审批"这种业务规则），这个确实可以配。但状态机的骨架不应该配。

## 九、单元测试

状态机天然适合单元测试，每个转移都可以独立测试：

```java
class SalesOrderStateMachineTest {

    private StateMachine<SalesOrderStatus, SalesOrderEvent> stateMachine;

    @BeforeEach
    void setUp() {
        // 构建不带条件和动作的纯状态机（只测流转合法性）
        stateMachine = StateMachine.<SalesOrderStatus, SalesOrderEvent>builder("TEST")
            .withTransition(Transition.from(DRAFT).to(CONFIRMED).on(SUBMIT).build())
            .withTransition(Transition.from(DRAFT).to(CANCELLED).on(CANCEL).build())
            .withTransition(Transition.from(CONFIRMED).to(PRODUCING).on(START_PRODUCE).build())
            .withTransition(Transition.from(CONFIRMED).to(CANCELLED).on(CANCEL).build())
            .withTransition(Transition.from(PRODUCING).to(READY).on(READY_STOCK).build())
            .withTransition(Transition.from(PRODUCING).to(SHIPPED).on(SHIP).build())
            .withTransition(Transition.from(READY).to(SHIPPED).on(SHIP).build())
            .withTransition(Transition.from(SHIPPED).to(COMPLETED).on(SIGN_OFF).build())
            .build();
    }

    @Test
    void should_allow_draft_to_pending_approval() {
        SalesOrderStatus result = stateMachine.fireEvent(DRAFT, SUBMIT, new TransitionContext<>());
        assertEquals(PENDING_APPROVAL, result);
    }

    @Test
    void should_reject_pending_approval_to_draft() {
        assertThrows(StateMachineException.class, () -> {
            stateMachine.fireEvent(PENDING_APPROVAL, SUBMIT, new TransitionContext<>());
        });
    }

    @Test
    void should_show_available_actions_for_draft() {
        List<Transition<SalesOrderStatus, SalesOrderEvent>> actions = 
            stateMachine.getAvailableTransitions(DRAFT);
        assertEquals(2, actions.size());  // SUBMIT 和 CANCEL
    }
}
```

## 十一、轻量审批引擎

状态机解决了业务单据的**状态流转**问题，审批引擎解决的是**谁有权触发流转**的问题。两者配合使用：状态机的 PENDING_APPROVAL 状态由审批引擎驱动进入 CONFIRMED 或 REJECTED。

### 11.1 需求

- 支持单人审批（如销售订单由业务经理审批）
- 支持多人或签（如采购订单由采购经理或采购总监任意一人审批即可）
- 审批意见必填
- 审批记录可追溯

不需要：条件分支（如金额>10万走不同路径）、会签（所有人都要批）、加签转签。

### 11.2 设计方案：轻量审批引擎

不引入Activiti/Flowable等重型流程引擎（功能过剩、学习成本高），自研轻量审批。

**核心模型：**

```
sys_approval_config（审批配置 — 定义什么业务需要谁审批）
├── id                       BIGPK          主键
├── business_type            VARCHAR(32)    业务类型
│                                          SALES_ORDER / PURCHASE_ORDER / 
│                                          QUANTITY_CHANGE / STOCKTAKING_DIFF
├── config_name              VARCHAR(64)    配置名称
├── approval_mode            VARCHAR(16)    SINGLE（单人）/ OR_SIGN（或签）
├── approver_role_ids        JSONB          审批人角色ID列表
│                                          单人模式：[roleId]
│                                          或签模式：[roleId1, roleId2, ...]
├── enabled                  BOOLEAN        是否启用
├── created_by / created_at / updated_by / updated_at / deleted / version
└──
```

**单人审批逻辑：** 审批配置中指定一个角色，该角色下任意一人审批即可通过。系统为该角色下每个用户生成一条待办，任一人审批后其他待办自动取消。

**或签逻辑：** 审批配置中指定多个角色，这些角色中任意一人审批即可通过。系统为每个符合条件的审批人生成一条待办，任一人审批后其他待办自动取消。

```
sys_approval_task（审批任务 — 每次审批生成）
├── id                       BIGPK          主键
├── business_type            VARCHAR(32)    业务类型
├── business_id              BIGINT         业务单据ID
├── business_no              VARCHAR(32)    业务单据编号
├── approval_mode            VARCHAR(16)    SINGLE / OR_SIGN
├── status                   VARCHAR(16)    PENDING / APPROVED / REJECTED / CANCELLED
├── approver_id              BIGINT         审批人ID（待办分配给谁）
├── approver_role_id         BIGINT         审批人角色ID
├── opinion                  TEXT           审批意见
├── approved_at              TIMESTAMP      审批时间
├── created_by / created_at / updated_by / updated_at / deleted / version
└──
```

### 11.3 审批流程代码

```java
@Service
public class ApprovalService {

    /**
     * 提交审批 — 为业务单据创建审批任务
     */
    @Transactional
    public void submitForApproval(String businessType, Long businessId, 
                                   String businessNo, Long submitterId) {
        // 1. 查询审批配置
        ApprovalConfig config = configRepo.findByBusinessType(businessType);
        if (config == null || !config.getEnabled()) {
            // 无审批配置 → 直接通过（自动审批）
            outboxRepo.save(DomainEvent.of("ApprovalPassed", businessId,
                Map.of("businessType", businessType, "businessId", businessId)));
            return;
        }

        // 2. 根据审批模式生成审批任务
        List<Long> approverRoleIds = config.getApproverRoleIds();
        
        if (config.getApprovalMode() == ApprovalMode.SINGLE) {
            Long roleId = approverRoleIds.get(0);
            for (Long approverId : userService.findUsersByRole(roleId)) {
                taskRepo.save(ApprovalTask.builder()
                    .businessType(businessType).businessId(businessId)
                    .businessNo(businessNo).approvalMode(SINGLE)
                    .status(PENDING).approverId(approverId)
                    .approverRoleId(roleId).build());
            }
        } else {
            for (Long roleId : approverRoleIds) {
                for (Long approverId : userService.findUsersByRole(roleId)) {
                    taskRepo.save(ApprovalTask.builder()
                        .businessType(businessType).businessId(businessId)
                        .businessNo(businessNo).approvalMode(OR_SIGN)
                        .status(PENDING).approverId(approverId)
                        .approverRoleId(roleId).build());
                }
            }
        }
    }

    /**
     * 审批操作
     */
    @Transactional
    public void approve(Long taskId, boolean approved, String opinion, Long operatorId) {
        ApprovalTask task = taskRepo.findById(taskId);
        
        if (!task.getApproverId().equals(operatorId)) {
            throw new BizException("无权审批此任务");
        }
        if (task.getStatus() != PENDING) {
            throw new BizException("任务已处理");
        }

        task.setStatus(approved ? APPROVED : REJECTED);
        task.setOpinion(opinion);
        task.setApprovedAt(LocalDateTime.now());
        taskRepo.save(task);

        // 单人审批和或签都可能有多条待办，任一人审批后取消同一业务的其他待办
        taskRepo.cancelOtherPendingTasks(
            task.getBusinessType(), task.getBusinessId(), task.getId());

        // 发布审批结果领域事件（各业务模块自行订阅处理，符合开闭原则）
        outboxRepo.save(DomainEvent.of(
            approved ? "ApprovalPassed" : "ApprovalRejected",
            task.getBusinessId(),
            Map.of("businessType", task.getBusinessType(),
                   "businessId", task.getBusinessId(),
                   "businessNo", task.getBusinessNo(),
                   "taskId", task.getId())
        ));
    }
}
```
